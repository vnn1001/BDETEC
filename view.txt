CREATE VIEW VW_LISTAALUNOS
AS SELECT Matricula,Nome,Data_Nascimento
FROM ALUNOS;

SELECT * FROM VW_LISTAALUNOS;

AS SELECT Matricula,Nome,Data_Nascimento
FROM ALUNOS
WHERE Nome='PEDRO ANTONIO SILVA';
